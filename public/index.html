// Show loading state function showLoading() { document.getElementById('news-container').innerHTML = `

Loading news...

Please wait while we fetch the latest articles

`; } // Show error state function showError(message) { document.getElementById('news-container').innerHTML = `

Something went wrong

${message}

`; } // Get category class for styling function getCategoryClass(category) { const normalizedCategory = category.toLowerCase(); return `category-${normalizedCategory}`; } // Display news articles function displayNews(articles) { const container = document.getElementById('news-container'); if (!articles || articles.length === 0) { container.innerHTML = `

No news articles found

We couldn't find any articles matching your criteria. Please try again later.

`; return; } // Sort articles by date (newest first) articles.sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp)); let newsHTML = ''; // Display up to 20 articles (including featured) const displayCount = Math.min(articles.length, 20); // Create featured article (first article) if we have enough articles if (displayCount > 0) { const featured = articles[0]; const category = featured.category.charAt(0).toUpperCase() + featured.category.slice(1); const categoryClass = getCategoryClass(featured.category); newsHTML += `
${category}

${featured.title}

${featured.source}
${featured.displayDate || new Date(featured.timestamp).toLocaleDateString()}
`; } // Create regular articles (remaining articles) for (let i = 1; i < displayCount; i++) { const article = articles[i]; const category = article.category.charAt(0).toUpperCase() + article.category.slice(1); const categoryClass = getCategoryClass(article.category); newsHTML += `
${category}

${article.title}

${article.source}
${article.displayDate || new Date(article.timestamp).toLocaleDateString()}
`; } container.innerHTML = newsHTML; } // Fetch all news async function fetchAllNews() { try { showLoading(); const response = await fetch('/api/news'); const data = await response.json(); if (data.status === 'success') { displayNews(data.data); document.getElementById('section-title').textContent = 'Latest News'; } else { showError('Failed to load news'); } } catch (error) { console.error('Error fetching news:', error); showError('Error loading news. Please try again later.'); } } // Fetch filtered news async function fetchFilteredNews(filter) { try { showLoading(); let response, data; switch(filter) { case 'singapore': response = await fetch('/api/news/singapore'); document.getElementById('section-title').textContent = 'Singapore News'; break; case 'malaysia': response = await fetch('/api/news/malaysia'); document.getElementById('section-title').textContent = 'Malaysia News'; break; case 'business': response = await fetch('/api/news/business'); document.getElementById('section-title').textContent = 'Business News'; break; case 'technology': response = await fetch('/api/news/technology'); document.getElementById('section-title').textContent = 'Technology News'; break; case 'sports': response = await fetch('/api/news/sports'); document.getElementById('section-title').textContent = 'Sports News'; break; case 'entertainment': response = await fetch('/api/news/entertainment'); document.getElementById('section-title').textContent = 'Entertainment News'; break; case 'health': response = await fetch('/api/news/health'); document.getElementById('section-title').textContent = 'Health News'; break; case 'science': response = await fetch('/api/news/science'); document.getElementById('section-title').textContent = 'Science News'; break; default: response = await fetch('/api/news'); document.getElementById('section-title').textContent = 'Latest News'; } data = await response.json(); if (data.status === 'success') { displayNews(data.data); } else { showError(`Failed to load ${filter} news`); } } catch (error) { console.error(`Error fetching ${filter} news:`, error); showError(`Error loading ${filter} news. Please try again later.`); } } // Search news async function searchNews(query) { try { showLoading(); const response = await fetch(`/api/news/search?q=${encodeURIComponent(query)}`); const data = await response.json(); if (data.status === 'success') { displayNews(data.data); document.getElementById('section-title').textContent = `Search Results for "${query}"`; } else { showError('Failed to search news'); } } catch (error) { console.error('Error searching news:', error); showError('Error searching news. Please try again later.'); } } // Initialize the application document.addEventListener('DOMContentLoaded', function() { console.log('QWEN-NEWS application loaded'); updateCurrentDate(); fetchAllNews(); // Add event listeners to navigation links document.querySelectorAll('.nav-link').forEach(link => { link.addEventListener('click', function(e) { e.preventDefault(); const filter = this.getAttribute('data-filter'); setActiveNav(filter); if (filter === 'all') { fetchAllNews(); } else { fetchFilteredNews(filter); } }); }); // Add event listeners to footer category links document.querySelectorAll('.footer-column ul li a[data-filter]').forEach(link => { link.addEventListener('click', function(e) { e.preventDefault(); const filter = this.getAttribute('data-filter'); // Update navigation setActiveNav(filter); // Update main content if (filter === 'all') { fetchAllNews(); } else { fetchFilteredNews(filter); } }); }); // Add event listener to search button document.getElementById('search-button').addEventListener('click', function() { const query = document.getElementById('search-input').value.trim(); if (query) { searchNews(query); } }); // Add event listener to search input for Enter key document.getElementById('search-input').addEventListener('keypress', function(e) { if (e.key === 'Enter') { const query = this.value.trim(); if (query) { searchNews(query); } } }); });
